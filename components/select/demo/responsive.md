## en-US

Responsive for multiple.

```tsx
import { Select } from 'antd';

const options: {
  label: string;
  value: string;
}[] = [];

for (let i = 10; i < 36; i++) {
  const value = i.toString(36) + i;
  options.push({
    label: `Long Label: ${value}`,
    value,
  });
}

ReactDOM.render(
  <>
    {/*<Select
      mode="multiple"
      style={{ width: '100%' }}
      defaultValue={['a10', 'c12']}
      options={options}
      placeholder="Select Item..."
      maxTagCount="responsive"
    />*/}

    <br />

    <Select
      open
      mode="multiple"
      style={{ width: '100%' }}
      defaultValue={['a10']}
      options={options}
      placeholder="Select Item..."
      maxTagCount="responsive"
    />
  </>,
  mountNode,
);
```
